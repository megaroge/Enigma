sub_dat <- filter(working_dat,
                   Product == "SINGULAIR" | 
                   Product == "TEMODAL"  | 
                   Product == "OTROS-RENITEC" |
                   Product == "OTROS-CORENITEC" | 
                   Product == "TIENAM" |
                   Product == "ONICIT" |
                   Product == "TEMGESIC")
p <- ggplot(sub_dat, aes(x=Month, y = Val, fill = Product))
p + geom_bar(stat = "identity", postition = "dodge") + scale_x_discrete()
